import streamlit as st
import pandas as pd
import plotly.express as px
from streamlit import caching
from datetime import datetime

# Set page configuration
st.set_page_config(
    page_title="Revenue Dashboard",
    page_icon="",
    layout="wide",
    initial_sidebar_state="expanded",
)

# Load dataset
@st.cache(allow_output_mutation=True)
def load_data(file):
    if file is not None:
        return pd.read_csv(file)
    else:
        # Use a sample dataset if no file is uploaded
        data = {
            "date": [
                datetime(2022, 1, 1),
                datetime(2022, 1, 2),
                datetime(2022, 1, 3),
                datetime(2022, 1, 4),
                datetime(2022, 1, 5),
            ],
            "revenue": [100, 120, 110, 130, 140],
            "users": [1000, 1100, 1200, 1300, 1400],
            "country": ["USA", "USA", "Canada", "Canada", "Mexico"],
        }
        return pd.DataFrame(data)

# Create a function to update the dataset
def update_dataset(df, file):
    if file is not None:
        return pd.read_csv(file)
    else:
        return df

# Create a function to filter the dataset
def filter_dataset(df, start_date, end_date, country):
    filtered_df = df[
        (df["date"] >= start_date) & (df["date"] <= end_date) & (df["country"] == country)
    ]
    return filtered_df

# Create a function to calculate KPIs
def calculate_kpis(df):
    total_revenue = df["revenue"].sum()
    total_users = df["users"].sum()
    average_revenue = df["revenue"].mean()
    average_users = df["users"].mean()
    return total_revenue, total_users, average_revenue, average_users

# Create a function to create plots
def create_plots(df):
    # Create a line chart for revenue
    revenue_fig = px.line(df, x="date", y="revenue", title="Revenue Over Time")
    revenue_fig.update_layout(xaxis_title="Date", yaxis_title="Revenue")

    # Create a line chart for users
    users_fig = px.line(df, x="date", y="users", title="Users Over Time")
    users_fig.update_layout(xaxis_title="Date", yaxis_title="Users")

    # Create a bar chart for revenue by country
    country_fig = px.bar(
        df.groupby("country")["revenue"].sum().reset_index(),
        x="country",
        y="revenue",
        title="Revenue by Country",
    )
    country_fig.update_layout(xaxis_title="Country", yaxis_title="Revenue")

    return revenue_fig, users_fig, country_fig

# Create a function to download the dataset
def download_dataset(df):
    return df.to_csv(index=False)

# Main app
def main():
    # Create a sidebar
    with st.sidebar:
        # Upload a CSV file
        file = st.file_uploader("Upload a CSV file", type=["csv"])

        # Create filters
        start_date = st.date_input("Start Date", min_value=datetime(2022, 1, 1), value=datetime(2022, 1, 1))
        end_date = st.date_input("End Date", min_value=datetime(2022, 1, 1), value=datetime(2022, 1, 31))
        country = st.selectbox("Country", ["USA", "Canada", "Mexico"])

        # Create a button to download the dataset
        with st.form("download_form"):
            submit_button = st.form_submit_button("Download CSV")

    # Load the dataset
    df = load_data(file)

    # Update the dataset
    df = update_dataset(df, file)

    # Filter the dataset
    filtered_df = filter_dataset(df, start_date, end_date, country)

    # Calculate KPIs
    total_revenue, total_users, average_revenue, average_users = calculate_kpis(filtered_df)

    # Create plots
    revenue_fig, users_fig, country_fig = create_plots(filtered_df)

    # Create a main container
    with st.container():
        # Create KPI cards
        col1, col2, col3, col4 = st.columns(4)
        col1.metric("Total Revenue", f"${total_revenue:,.2f}")
        col2.metric("Total Users", f"{total_users:,.2f}")
        col3.metric("Average Revenue", f"${average_revenue:,.2f}")
        col4.metric("Average Users", f"{average_users:,.2f}")

        # Create a row for plots
        st.row(
            [
                st.plotly_chart(revenue_fig, use_container_width=True),
                st.plotly_chart(users_fig, use_container_width=True),
            ]
        )

        # Create a row for the country plot
        st.plotly_chart(country_fig, use_container_width=True)

        # Download the dataset
        if submit_button:
            st.download_button("Download CSV", download_dataset(filtered_df), "data.csv", "text/csv")

if __name__ == "__main__":
    main()
